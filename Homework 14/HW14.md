     s <- rbinom(2*k-1,1,p_A)
     o <- ifelse(sum(s) > k - 1, "A", "B")
     n1 <- cumsum(s)
     n2 <- cumsum(1-s)

    }
      return(list(o = o, n = n))

    # 6 

    rsimulate <- function(R, k, p_A){
     o <- rep(NA, R) 
     n <- rep(NA, R) 
     for (i in 1:R) {
        result <- simulate(k, p_A)
        o[i] <- result$o
        n[i] <- result$n
     }
     return(data.frame(outcome = o, number_games = n))
    }
